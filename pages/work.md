<script>
    function revealSection(sectionId, btn) {
        var section = document.getElementById(sectionId);
        var isHidden = section.style.display === 'none';
        
        if (isHidden) {
            // SHOW
            section.style.display = 'block';
            btn.innerHTML = btn.innerHTML.replace('View', 'Hide').replace('↓', '↑');
            btn.classList.add('active-btn');
            
            // Smoothly scroll to the newly revealed section
            setTimeout(() => {
                section.scrollIntoView({ behavior: 'smooth', block: 'start' });
            }, 50);
        } else {
            // HIDE
            section.style.display = 'none';
            btn.innerHTML = btn.innerHTML.replace('Hide', 'View').replace('↑', '↓');
            btn.classList.remove('active-btn');
            
            // Scroll back up to the intro or button area if needed
            document.querySelector('.work-intro').scrollIntoView({ behavior: 'smooth' });
        }
    }
</script>
